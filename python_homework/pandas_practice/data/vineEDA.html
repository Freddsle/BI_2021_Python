      def read_gff(file):
          '''
          Read gff file (input - path to the file) and return Pandas DataFrame.
          Do not works with really large gff files.
          '''    
          colnames = ['chromosome', 'source', 'type',
                      'start', 'end', 'score',
                      'strand', 'phase', 'attributes']
          df_gff = pd.read_csv(file,
                               sep='\t',
                               engine='python',
                               header=None,
                               names=colnames,
                               comment='#')
          
          return df_gff
